\subsection{Intersección}

\lipsum[2-3]

\separation
\begin{definitionblock}{Intersección}
    $A \cap B = \{x : x \in A \wedge x \in B \}$
\end{definitionblock}

\separation
\lipsum[2-3]

\separation
\begin{exampleblock}{Intersección}
    \begin{equation*}
        A = \{1,3,7\} \\
        B = \{10,11,1,3\} \\
        A \cap B = \{1,3\}
    \end{equation*}
\end{exampleblock}
