\newenvironment{definitionblock}[1]{%
	\tcolorbox[beamer,%
    noparskip,breakable,%
    colback=WhiteSmoke,colframe=Olive,%
    colbacklower=Olive!75!WhiteSmoke,%
	title=\definition{#1}]}%
	{\endtcolorbox}
	
\newenvironment{exampleblock}[1]{%
    \tcolorbox[beamer,%
    noparskip,breakable,%
    colback=WhiteSmoke,colframe=DarkGreen,%
    colbacklower=LimeGreen!75!WhiteSmoke,%
    title=\example{#1}]}%
    {\endtcolorbox}

\newcommand{\listdefinitionname}{Índice de Definiciones}

\newcommand{\definition}[1]{%
\refstepcounter{definition}
\par\noindent\textbf{Definición \thedefinition.} \whiteref{#1}
\addcontentsline{def}{definition}{\protect\numberline{\thedefinition}#1}\par}
